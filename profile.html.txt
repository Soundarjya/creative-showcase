<h2 id="profileName"></h2>
<div class="gallery" id="profileGallery"></div>

<script>
const user = new URLSearchParams(window.location.search).get("user");
document.getElementById("profileName").innerText = user + "'s Gallery";

const images = JSON.parse(localStorage.getItem("images_" + user)) || [];
const gallery = document.getElementById("profileGallery");

images.forEach(img => {
  const image = document.createElement("img");
  image.src = img;
  gallery.appendChild(image);
});
</script>
